## Overview

VW Phones is a full-featured e-commerce platform designed for smartphone retail. The application simulates a complete online store experience with robust inventory management, product catalog browsing, and administrative controls for managing stock levels, product information, and sales.

Built with Flask's lightweight framework, the project demonstrates efficient backend architecture for e-commerce operations while maintaining a clean, user-friendly frontend interface.

### Database Schema

<Tabs>
  <Tab title="Database Structure">
    ```sql
    -- Products table: Core smartphone information
    CREATE TABLE products (
        id INT PRIMARY KEY AUTO_INCREMENT,
        model_name VARCHAR(200) NOT NULL,
        brand VARCHAR(50) NOT NULL,
        price DECIMAL(10, 2) NOT NULL,
        description TEXT,
        screen_size DECIMAL(3, 1),
        storage_capacity INT,
        ram INT,
        camera_mp VARCHAR(50),
        battery_mah INT,
        color VARCHAR(50),
        stock_quantity INT DEFAULT 0,
        featured BOOLEAN DEFAULT FALSE,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
    );
    
    -- Product images table: Multiple photos per device
    CREATE TABLE product_images (
        id INT PRIMARY KEY AUTO_INCREMENT,
        product_id INT NOT NULL,
        image_url VARCHAR(500) NOT NULL,
        is_primary BOOLEAN DEFAULT FALSE,
        display_order INT DEFAULT 0,
        FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE
    );
    
    -- Inventory transactions: Stock movement history
    CREATE TABLE inventory_transactions (
        id INT PRIMARY KEY AUTO_INCREMENT,
        product_id INT NOT NULL,
        transaction_type ENUM('stock_in', 'stock_out', 'adjustment'),
        quantity INT NOT NULL,
        reason VARCHAR(200),
        user_id INT,
        transaction_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        FOREIGN KEY (product_id) REFERENCES products(id)
    );
    
    -- Shopping cart
    CREATE TABLE cart_items (
        id INT PRIMARY KEY AUTO_INCREMENT,
        session_id VARCHAR(100) NOT NULL,
        product_id INT NOT NULL,
        quantity INT DEFAULT 1,
        added_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        FOREIGN KEY (product_id) REFERENCES products(id)
    );
    ```

    ```sql
    -- Stored procedure for stock adjustment with logging
    DELIMITER //
    CREATE PROCEDURE adjust_stock(
        IN p_product_id INT,
        IN p_quantity INT,
        IN p_transaction_type VARCHAR(20),
        IN p_reason VARCHAR(200)
    )
    BEGIN
        DECLARE current_stock INT;
        
        -- Start transaction
        START TRANSACTION;
        
        -- Get current stock
        SELECT stock_quantity INTO current_stock
        FROM products
        WHERE id = p_product_id
        FOR UPDATE;
        
        -- Validate stock availability for stock_out
        IF p_transaction_type = 'stock_out' AND current_stock < p_quantity THEN
            SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'Insufficient stock';
        END IF;
        
        -- Update stock quantity
        IF p_transaction_type = 'stock_in' THEN
            UPDATE products
            SET stock_quantity = stock_quantity + p_quantity
            WHERE id = p_product_id;
        ELSE
            UPDATE products
            SET stock_quantity = stock_quantity - p_quantity
            WHERE id = p_product_id;
        END IF;
        
        -- Log transaction
        INSERT INTO inventory_transactions
        (product_id, transaction_type, quantity, reason)
        VALUES (p_product_id, p_transaction_type, p_quantity, p_reason);
        
        COMMIT;
    END //
    DELIMITER ;
    ```

    from flask import Flask, render_template, request, redirect, url_for, session
    from flask_mysqldb import MySQL
    
    app = Flask(__name__)
    mysql = MySQL(app)
    
    @app.route('/')
    def index():
        """Homepage with featured products"""
        cursor = mysql.connection.cursor()
        cursor.execute('''
            SELECT p.*, pi.image_url
            FROM products p
            LEFT JOIN product_images pi ON p.id = pi.product_id AND pi.is_primary = 1
            WHERE p.featured = 1 AND p.stock_quantity > 0
            LIMIT 6
        ''')
        featured_products = cursor.fetchall()
        cursor.close()
        return render_template('index.html', products=featured_products)
    
    @app.route('/catalog')
    def catalog():
        """Product catalog with filters"""
        brand = request.args.get('brand', '')
        min_price = request.args.get('min_price', 0)
        max_price = request.args.get('max_price', 99999)
        
        query = '''
            SELECT p.*, pi.image_url
            FROM products p
            LEFT JOIN product_images pi ON p.id = pi.product_id AND pi.is_primary = 1
            WHERE p.stock_quantity > 0
            AND p.price BETWEEN %s AND %s
        '''
        params = [min_price, max_price]
        
        if brand:
            query += ' AND p.brand = %s'
            params.append(brand)
            
        query += ' ORDER BY p.brand, p.model_name'
        
        cursor = mysql.connection.cursor()
        cursor.execute(query, params)
        products = cursor.fetchall()
        cursor.close()
        
        return render_template('catalog.html', products=products)
    
    @app.route('/product/<int:product_id>')
    def product_detail(product_id):
        """Detailed product page"""
        cursor = mysql.connection.cursor()
        
        # Get product details
        cursor.execute('SELECT * FROM products WHERE id = %s', [product_id])
        product = cursor.fetchone()
        
        # Get all images
        cursor.execute('''
            SELECT image_url FROM product_images
            WHERE product_id = %s
            ORDER BY display_order
        ''', [product_id])
        images = cursor.fetchall()
        
        cursor.close()
        return render_template('product.html', product=product, images=images)
    
    @app.route('/admin/inventory')
    def admin_inventory():
        """Admin inventory management"""
        cursor = mysql.connection.cursor()
        cursor.execute('''
            SELECT id, model_name, brand, price, stock_quantity
            FROM products
            ORDER BY brand, model_name
        ''')
        products = cursor.fetchall()
        cursor.close()
        return render_template('admin/inventory.html', products=products)
    
    @app.route('/admin/product/add', methods=['GET', 'POST'])
    def admin_add_product():
        """Add new product"""
        if request.method == 'POST':
            cursor = mysql.connection.cursor()
            cursor.execute('''
                INSERT INTO products
                (model_name, brand, price, description, screen_size,
                 storage_capacity, ram, camera_mp, battery_mah, color, stock_quantity)
                VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s)
            ''', [
                request.form['model_name'],
                request.form['brand'],
                request.form['price'],
                request.form['description'],
                request.form['screen_size'],
                request.form['storage_capacity'],
                request.form['ram'],
                request.form['camera_mp'],
                request.form['battery_mah'],
                request.form['color'],
                request.form['stock_quantity']
            ])
            mysql.connection.commit()
            cursor.close()
            return redirect(url_for('admin_inventory'))
        
        return render_template('admin/add_product.html')
    ```

    @app.route('/admin/stock/adjust/<int:product_id>', methods=['POST'])
    def adjust_stock(product_id):
        """Adjust product stock levels"""
        quantity = int(request.form['quantity'])
        transaction_type = request.form['transaction_type']
        reason = request.form.get('reason', '')
        
        cursor = mysql.connection.cursor()
        
        try:
            # Call stored procedure
            cursor.callproc('adjust_stock', [
                product_id,
                quantity,
                transaction_type,
                reason
            ])
            mysql.connection.commit()
            
            flash('Stock adjusted successfully', 'success')
        except Exception as e:
            mysql.connection.rollback()
            flash(f'Error adjusting stock: {str(e)}', 'error')
        finally:
            cursor.close()
        
        return redirect(url_for('admin_inventory'))
    
    @app.route('/admin/stock/low-stock')
    def low_stock_report():
        """View products with low stock"""
        cursor = mysql.connection.cursor()
        cursor.execute('''
            SELECT model_name, brand, stock_quantity, price
            FROM products
            WHERE stock_quantity < 5 AND stock_quantity > 0
            ORDER BY stock_quantity ASC
        ''')
        low_stock_items = cursor.fetchall()
        cursor.close()
        
        return render_template('admin/low_stock.html', items=low_stock_items)

    @app.route('/cart/add/<int:product_id>', methods=['POST'])
    def add_to_cart(product_id):
        """Add product to shopping cart"""
        quantity = int(request.form.get('quantity', 1))
        
        # Get or create session ID
        if 'session_id' not in session:
            import uuid
            session['session_id'] = str(uuid.uuid4())
        
        cursor = mysql.connection.cursor()
        
        # Check stock availability
        cursor.execute(
            'SELECT stock_quantity FROM products WHERE id = %s',
            [product_id]
        )
        result = cursor.fetchone()
        
        if result and result[0] >= quantity:
            # Add to cart or update quantity
            cursor.execute('''
                INSERT INTO cart_items (session_id, product_id, quantity)
                VALUES (%s, %s, %s)
                ON DUPLICATE KEY UPDATE quantity = quantity + %s
            ''', [session['session_id'], product_id, quantity, quantity])
            mysql.connection.commit()
            flash('Product added to cart', 'success')
        else:
            flash('Insufficient stock', 'error')
        
        cursor.close()
        return redirect(url_for('product_detail', product_id=product_id))
    
    @app.route('/cart')
    def view_cart():
        """View shopping cart"""
        if 'session_id' not in session:
            return render_template('cart.html', items=[], total=0)
        
        cursor = mysql.connection.cursor()
        cursor.execute('''
            SELECT p.id, p.model_name, p.brand, p.price,
                   ci.quantity, (p.price * ci.quantity) as subtotal,
                   pi.image_url
            FROM cart_items ci
            JOIN products p ON ci.product_id = p.id
            LEFT JOIN product_images pi ON p.id = pi.product_id AND pi.is_primary = 1
            WHERE ci.session_id = %s
        ''', [session['session_id']])
        items = cursor.fetchall()
        
        total = sum(item[5] for item in items)  # Sum of subtotals
        
        cursor.close()
        return render_template('cart.html', items=items, total=total)
    ```
